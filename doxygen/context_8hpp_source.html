264 [[nodiscard]] auto new_console(int min_columns = 1, int min_rows = 1, float magnification = 1.0f) -> tcod::Console {
265 int columns;
266 int rows;
267 if (magnification <= 0.0f) {
268 throw std::invalid_argument(
269 std::string("Magnification must be greater than zero. Got ") + std::to_string(magnification));
270 }
271 tcod::check_throw_error(TCOD_context_recommended_console_size(context_.get(), magnification, &columns, &rows));
272 return tcod::Console{std::max(columns, min_columns), std::max(rows, min_rows)};
273 }
