194template <typename... T>
195inline std::string stringf(const char* format, T... args) {
196 const int str_length = snprintf(nullptr, 0, format, args...);
197 if (str_length < 0) throw std::runtime_error("Failed to format string.");
198 std::string out(str_length, '\0');
199 snprintf(&out[0], str_length + 1, format, args...);
200 return out;
201}
