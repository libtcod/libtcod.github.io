494 auto new_console(int min_columns = 1, int min_rows = 1, float magnification = 1.0f) -> tcod::Console {
495 int columns;
496 int rows;
497 if (magnification <= 0.0f) {

500 }
501 tcod::check_throw_error(TCOD_context_recommended_console_size(this, magnification, &columns, &rows));
502 return tcod::Console{std::max(columns, min_columns), std::max(rows, min_rows)};
