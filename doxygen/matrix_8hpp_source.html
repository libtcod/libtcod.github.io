42template <typename ArrayType>
43[[nodiscard]] static std::string array_as_string(const ArrayType& arr) {
44 std::string result{"{"};
45 for (const auto& it : arr) {
46 result += std::to_string(it);
47 if (&it != &arr.back()) {
48 result += ", ";
49 }
50 }
51 result += "}";
52 return result;
53}

63template <typename T, size_t Dimensions>
64class MatrixView {
65 public:
66 using size_type = int; // The int size of indexes.
67 using shape_type = std::array<size_type, Dimensions>; // The type used to store the matrixes shape.
68 using stride_type = std::array<size_type, Dimensions>; // The type used to store strides.
69 using index_type = std::array<size_type, Dimensions>; // The type used to index the container.
70 using reference = T&;
71 using const_reference = const T&;
73 constexpr MatrixView() = default;
75 constexpr MatrixView(const shape_type& shape_xy, const stride_type& strides_xy, T* data) noexcept
76 : shape_xy_{shape_xy}, strides_xy_{strides_xy}, data_{reinterpret_cast<data_ptr>(data)} {};
77
79 [[nodiscard]] constexpr reference operator[](const index_type& index) noexcept {
80 return *reinterpret_cast<T*>(data_ + get_offset(index));
81 }
83 [[nodiscard]] constexpr const_reference operator[](const index_type& index) const noexcept {
84 return *reinterpret_cast<const T*>(data_ + get_offset(index));
85 }
87 [[nodiscard]] constexpr reference at(const index_type& index) {
88 return *reinterpret_cast<T*>(data_ + check_range(index));
89 }
91 [[nodiscard]] constexpr const_reference at(const index_type& index) const {
92 return *reinterpret_cast<const T*>(data_ + check_range(index));
93 }
94
96 [[nodiscard]] constexpr bool in_bounds(const index_type& index) const noexcept {
97 for (size_t dimension = 0; dimension < Dimensions; ++dimension) {
98 if (!(0 <= index.at(dimension) && index.at(dimension) < shape_xy_.at(dimension))) return false;
99 }
100 return true;
101 }
102
103 private:
104 // `unsigned char*` pointer which has the same const as `T`.
105 using data_ptr = typename std::conditional<std::is_const<T>::value, const unsigned char*, unsigned char*>::type;
107 [[nodiscard]] constexpr size_t get_offset(const index_type& index) const noexcept {
108 size_t data_index = 0;
109 for (size_t dimension{0}; dimension < Dimensions; ++dimension) {
110 data_index += strides_xy_.at(dimension) * index.at(dimension);
111 }
112 return data_index;
113 }
115 constexpr size_t check_range(const index_type& index) const {
116 using internal::array_as_string;
117 if (!in_bounds(index)) {
118 throw std::out_of_range(
119 std::string("Out of bounds lookup ") + array_as_string(index) + " on matrix of shape " +
120 array_as_string(shape_xy_) + ".");
121 }
122 return get_offset(index);
123 }
124 shape_type shape_xy_; // The shape of this view.
125 stride_type strides_xy_; // The strides of this view in bytes.
126 data_ptr data_; // A pointer to the viewed memory.
127};

137template <typename T, size_t Dimensions, typename Container = std::vector<T>>
138class Matrix {
139 public:
140 using size_type = int; // The int size of indexes.
141 using shape_type = std::array<size_type, Dimensions>; // The type used to measure the matrixes shape.
142 using index_type = std::array<size_type, Dimensions>; // The type used to index the container.
143 using reference = typename Container::reference;
144 using const_reference = typename Container::const_reference;
146 constexpr Matrix() = default;
147
149 constexpr explicit Matrix(const shape_type& shape) : shape_{shape}, data_(get_size_from_shape(shape)) {}
150
152 constexpr Matrix(const shape_type& shape, const T& fill_value)
153 : shape_{shape}, data_(get_size_from_shape(shape), fill_value) {}
154
156 [[nodiscard]] constexpr auto begin() noexcept { return data_.begin(); }
158 [[nodiscard]] constexpr auto begin() const noexcept { return data_.cbegin(); }
159
161 [[nodiscard]] constexpr auto end() noexcept { return data_.end(); }
163 [[nodiscard]] constexpr auto end() const noexcept { return data_.cend(); }
164
166 [[nodiscard]] constexpr reference operator[](const index_type& index) noexcept { return data_[get_index(index)]; }
167
169 [[nodiscard]] constexpr const_reference operator[](const index_type& index) const noexcept {
170 return data_[get_index(index)];
171 }
173 [[nodiscard]] constexpr reference at(const index_type& index) { return data_.at(check_range(index)); }
174
176 [[nodiscard]] constexpr const_reference at(const index_type& index) const { return data_.at(check_range(index)); }
177
179 [[nodiscard]] constexpr const shape_type& get_shape() const noexcept { return shape_; }
180
182 [[nodiscard]] constexpr bool in_bounds(const index_type& index) const noexcept {
183 for (size_t dimension = 0; dimension < Dimensions; ++dimension) {
184 if (!(0 <= index.at(dimension) && index.at(dimension) < shape_.at(dimension))) return false;
185 }
186 return true;
187 }
188
190 [[nodiscard]] constexpr operator MatrixView<T, Dimensions>() noexcept {
191 return {get_shape(), get_strides(), data_.data()};
192 }
194 [[nodiscard]] constexpr operator MatrixView<const T, Dimensions>() const noexcept {
195 return {get_shape(), get_strides(), data_.data()};
196 }
197
199 [[nodiscard]] constexpr Container& get_container() noexcept { return data_; }
200
202 [[nodiscard]] constexpr const Container& get_container() const noexcept { return data_; }
203
204 template <class Archive>
205 void serialize(Archive& archive) {
206 archive(shape_, data_);
207 }
208
209 private:
211 [[nodiscard]] static constexpr size_t get_size_from_shape(const shape_type& shape) noexcept {
212 size_t size = 1;
213 for (auto& it : shape) size *= it;
214 return size;
215 }
217 [[nodiscard]] constexpr size_t get_index(const index_type& index) const noexcept {
218 size_t stride = 1;
219 size_t data_index = 0;
220 for (size_t dimension = 0; dimension < Dimensions; ++dimension) {
221 data_index += stride * index.at(dimension);
222 stride *= shape_.at(dimension);
223 }
224 return data_index;
225 }
227 constexpr size_t check_range(const index_type& index) const {
228 using internal::array_as_string;
229 if (!in_bounds(index)) {
230 throw std::out_of_range(
231 std::string("Out of bounds lookup ") + array_as_string(index) + " on matrix of shape " +
232 array_as_string(shape_) + ".");
233 }
234 return get_index(index);
235 }
237 [[nodiscard]] constexpr index_type get_strides() const noexcept {
238 index_type strides{};
239 int stride = static_cast<int>(sizeof(T));
240 for (size_t dimension = 0; dimension < Dimensions; ++dimension) {
241 strides.at(dimension) = stride;
242 stride *= shape_.at(dimension);
243 }
244 return strides;
245 }
246 shape_type shape_;
247 Container data_;
248};
