312 inline Matrix get_matrix() const noexcept { return get_matrix(origin_, dest_); }
313 static Matrix get_matrix(const Point2& origin, const Point2& dest) noexcept {
314 const int delta_x = dest.at(0) - origin.at(0);
315 const int delta_y = dest.at(1) - origin.at(1);
316 Matrix matrix{
317 origin.at(0),
318 origin.at(1),
319 1,
320 0,
321 0,
322 1,
323 };
324 if (delta_x < 0) matrix.xx = -1;
325 if (delta_y < 0) matrix.yy = -1;
326 if (std::abs(delta_y) > std::abs(delta_x)) {
327 std::swap(matrix.xx, matrix.yx);
328 std::swap(matrix.xy, matrix.yy);
329 }
330 return matrix;
331 }

339 inline Point2 get_normalized_delta() const noexcept { return get_normalized_delta(origin_, dest_); }
340 static Point2 get_normalized_delta(const Point2& origin, const Point2& dest) noexcept {
341 return std::abs(dest.at(0) - origin.at(0)) > std::abs(dest.at(1) - origin.at(1))
342 ? Point2{std::abs(dest.at(0) - origin.at(0)), std::abs(dest.at(1) - origin.at(1))}
343 : Point2{std::abs(dest.at(1) - origin.at(1)), std::abs(dest.at(0) - origin.at(0))};
344 }
350 inline int get_delta_x() const noexcept { return get_normalized_delta().at(0); }
354 inline BresenhamLine& bresenham_next() {
355 const Point2 delta = get_normalized_delta();
356 y_error_ += delta.at(1);
357 if (y_error_ > 0) {
358 ++cursor_.at(1);
359 y_error_ -= delta.at(0);
360 };
361 ++cursor_.at(0);
362 return *this;
363 }
367 inline BresenhamLine& bresenham_prev() {
368 const Point2 delta = get_normalized_delta();
369 y_error_ -= delta.at(1);
370 if (y_error_ <= -delta.at(0)) {
371 --cursor_.at(1);
372 y_error_ += delta.at(0);
373 };
374 --cursor_.at(0);
375 return *this;
376 }
380 inline Point2 bresenham_get(int index) {
381 while (cursor_.at(0) < index) bresenham_next();
382 while (cursor_.at(0) > index) bresenham_prev();
383 return get_matrix().transform(cursor_);
384 }
385 Point2 origin_; // Starting point.
386 Point2 dest_; // Ending point.
387 int index_; // The starting index returned by `begin`.
388 int index_end_; // The past-the-end index returned by `end`.
389 Point2 cursor_; // Normalized Bresenham low-slope position. First axis acts as the current index.
390 int y_error_; // Fractional difference between Y indexes. Is always `-delta[0] < err <= 0`.
391};
