583 template <int OutSize, typename KeyColors, typename KeyIndexes>
584 [[nodiscard]] static constexpr auto genMap(const KeyColors& key_colors, const KeyIndexes& key_indexes)
585 -> std::array<tcod::ColorRGB, OutSize> {
586 auto out = std::array<tcod::ColorRGB, OutSize>{};
587 if (key_colors.size() != key_indexes.size())
588 throw std::invalid_argument{"Key color and index arrays must be the same size."};
589 for (size_t segment = 1; segment < key_colors.size(); ++segment) { // Segment between two key colors/indexes.
590 const auto index_start = key_indexes.at(segment - 1);
591 const auto index_end = key_indexes.at(segment);
592 if (segment == 1 && index_start != 0) throw std::invalid_argument{"First key must be 0."};
593 if (segment == key_colors.size() - 1 && index_end != OutSize - 1) {
594 throw std::invalid_argument{"Last key must be at the end of the output size."};
595 }
596 if (!(index_start < index_end)) throw std::invalid_argument{"Key indexes must be in ascending order."};
597 const auto color_start = TCODColor{key_colors.at(segment - 1)};
598 const auto color_end = TCODColor{key_colors.at(segment)};
599 for (auto i = index_start; i <= index_end; ++i) {
600 const float interpolation = static_cast<float>(i - index_start) / (index_end - index_start);
601 out.at(i) = tcod::ColorRGB{TCODColor::lerp(color_start, color_end, interpolation)};
602 }
603 }
604 return out;
605 }
