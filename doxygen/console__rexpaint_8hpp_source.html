58inline std::vector<tcod::ConsolePtr> load_xp(const std::filesystem::path& path) {
59 const auto path_str = path.string();
60 int layer_count = tcod::check_throw_error(TCOD_load_xp(path_str.c_str(), 0, nullptr));
61 std::vector<TCOD_Console*> tmp(layer_count, nullptr);
62 tcod::check_throw_error(TCOD_load_xp(path_str.c_str(), layer_count, &tmp[0]));
63 return std::vector<tcod::ConsolePtr>(tmp.begin(), tmp.end());
64}
